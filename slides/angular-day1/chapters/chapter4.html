            {
              path: 'admin',
              loadChildren: './views/admin/admin.module#AdminModule'
            },
            {
              path: 'user',
              loadChildren: './views/user/user.module#UserModule'
            },
            {
              path: 'home',
              loadChildren: './views/home/home.module#HomeModule'
            },
            {
              path: '',
              redirectTo: 'home',
              pathMatch: 'full'
